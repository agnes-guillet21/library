DROP TABLE IF EXISTS books;
 
CREATE TABLE books (
  id INT AUTO_INCREMENT  PRIMARY KEY,
  title VARCHAR(250) NOT NULL,
  description VARCHAR(250) NULL,
  author(250) NOT NULL,
  isbn INT NOT NULL
);
